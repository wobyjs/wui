## Shadow DOM Piercing Pattern (used in every test)

Every test script begins with this setup block to pierce the shadow root per D-12:

```javascript
const host = document.querySelector('wui-editor')
const sr = host.shadowRoot
const editorDiv = sr.querySelector('[data-editor-root]')
```

All queries into editor content use `sr.querySelector(...)` or `editorDiv.querySelector(...)`, never the flat `document.querySelector(...)`.

## Helper Injection (run once before all tests)

Before running individual tests, inject this helper via a single `evaluate_script` call:

```javascript
window._testHelper = {
    getEditor() {
        const host = document.querySelector('wui-editor')
        const sr = host.shadowRoot
        const editorDiv = sr.querySelector('[data-editor-root]')
        return { host, sr, editorDiv }
    },

    // Reset editor to a single paragraph using DOM methods only
    resetEditor(text) {
        const { editorDiv } = this.getEditor()
        while (editorDiv.firstChild) editorDiv.removeChild(editorDiv.firstChild)
        const p = document.createElement('p')
        p.textContent = text
        editorDiv.appendChild(p)
        return p
    },

    // Create two-paragraph editor state
    resetEditorTwoParagraphs(text1, text2) {
        const { editorDiv } = this.getEditor()
        while (editorDiv.firstChild) editorDiv.removeChild(editorDiv.firstChild)
        const p1 = document.createElement('p')
        p1.textContent = text1
        const p2 = document.createElement('p')
        p2.textContent = text2
        editorDiv.appendChild(p1)
        editorDiv.appendChild(p2)
        return { p1, p2 }
    },

    selectRange(textNode, start, end) {
        const range = document.createRange()
        range.setStart(textNode, start)
        range.setEnd(textNode, end)
        const sel = window.getSelection()
        sel.removeAllRanges()
        sel.addRange(range)
        return sel
    },

    selectContents(el) {
        const range = document.createRange()
        range.selectNodeContents(el)
        const sel = window.getSelection()
        sel.removeAllRanges()
        sel.addRange(range)
        return sel
    },

    getBoldButton() {
        const { sr } = this.getEditor()
        return sr.querySelector('[title="Bold"]') || document.querySelector('[title="Bold"]')
    },

    getItalicButton() {
        const { sr } = this.getEditor()
        return sr.querySelector('[title="Italic"]') || document.querySelector('[title="Italic"]')
    },

    clickButton(btn) {
        if (!btn) { console.warn('[testHelper] button not found'); return }
        btn.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true, composed: true }))
        btn.dispatchEvent(new MouseEvent('click', { bubbles: true, cancelable: true, composed: true }))
    },

    clickBold() { this.clickButton(this.getBoldButton()) },
    clickItalic() { this.clickButton(this.getItalicButton()) },

    activateEditor() {
        const { editorDiv } = this.getEditor()
        editorDiv.dispatchEvent(new MouseEvent('click', { bubbles: true, composed: true }))
    },

    hasBoldInElement(el) {
        const walker = document.createTreeWalker(el, NodeFilter.SHOW_ELEMENT)
        let node
        while ((node = walker.nextNode())) {
            const fw = window.getComputedStyle(node).fontWeight
            if ((fw === '700' || fw === 'bold') && node.textContent.trim()) return true
        }
        const fw = window.getComputedStyle(el).fontWeight
        return fw === '700' || fw === 'bold'
    },

    isAllBoldRemoved(el) {
        if (!el.querySelector('strong, b')) {
            const styledSpans = el.querySelectorAll('[style]')
            let hasBoldSpan = false
            styledSpans.forEach(s => {
                const fw = window.getComputedStyle(s).fontWeight
                if ((fw === '700' || fw === 'bold') && s.textContent.trim()) hasBoldSpan = true
            })
            if (!hasBoldSpan) return true
        }
        return false
    },

    assert(name, condition, failReason) {
        console.log('[TEST] ' + name + ': ' + (condition ? 'PASS' : 'FAIL - ' + failReason))
    }
}
console.log('[TEST] helpers-injected: PASS')
```

Verify via `list_console_messages()` that `[TEST] helpers-injected: PASS` appears before proceeding to task tests.

### Task 3.3: Toggle sequence tests (Dimension 4)

**File**: No source file — MCP `evaluate_script` tests only.

**Decision**: D-09

**Description**:
Verify MS Word / Google Docs toggle semantics: Bold 1x on, 2x off, 3x on. Bold + Italic = both. Bold + Italic + Bold = italic only.

These tests directly prove D-03 semantics work through the D-02 fix.

**Test 3.3-a: Bold toggle cycle — 1x on, 2x off, 3x on**:

```javascript
const p = window._testHelper.resetEditor('Toggle test')
window._testHelper.activateEditor()

function selectAll() { window._testHelper.selectContents(p) }

selectAll()
window._testHelper.clickBold()

setTimeout(() => {
    const after1 = window._testHelper.hasBoldInElement(p)
    window._testHelper.assert('3.3a-bold-1x-on', after1, 'bold not applied. outerHTML=' + p.outerHTML)

    selectAll()
    window._testHelper.clickBold()

    setTimeout(() => {
        const after2 = window._testHelper.isAllBoldRemoved(p)
        window._testHelper.assert('3.3a-bold-2x-off', after2, 'bold not removed. outerHTML=' + p.outerHTML)

        selectAll()
        window._testHelper.clickBold()

        setTimeout(() => {
            const after3 = window._testHelper.hasBoldInElement(p)
            window._testHelper.assert('3.3a-bold-3x-on', after3, 'bold not re-applied. outerHTML=' + p.outerHTML)
        }, 100)
    }, 100)
}, 100)
```

**Test 3.3-b: Bold then Italic — both styles merged**:

```javascript
const p = window._testHelper.resetEditor('Bold and italic')
window._testHelper.activateEditor()

window._testHelper.selectContents(p)
window._testHelper.clickBold()

setTimeout(() => {
    window._testHelper.selectContents(p)
    window._testHelper.clickItalic()

    setTimeout(() => {
        let hasBold = false, hasItalic = false
        const walker = document.createTreeWalker(p, NodeFilter.SHOW_ELEMENT)
        let node
        while ((node = walker.nextNode())) {
            const cs = window.getComputedStyle(node)
            if ((cs.fontWeight === '700' || cs.fontWeight === 'bold') && node.textContent.trim()) hasBold = true
            if (cs.fontStyle === 'italic' && node.textContent.trim()) hasItalic = true
        }
        window._testHelper.assert('3.3b-bold-italic-merged', hasBold && hasItalic,
            'hasBold=' + hasBold + ' hasItalic=' + hasItalic + '. outerHTML=' + p.outerHTML)
    }, 100)
}, 100)
```

**Test 3.3-c: Bold then Italic then Bold — only italic remains**:

```javascript
const p = window._testHelper.resetEditor('Sequential toggle')
window._testHelper.activateEditor()

window._testHelper.selectContents(p)
window._testHelper.clickBold()

setTimeout(() => {
    window._testHelper.selectContents(p)
    window._testHelper.clickItalic()

    setTimeout(() => {
        window._testHelper.selectContents(p)
        window._testHelper.clickBold()

        setTimeout(() => {
            let hasBold = false, hasItalic = false
            const walker = document.createTreeWalker(p, NodeFilter.SHOW_ELEMENT)
            let node
            while ((node = walker.nextNode())) {
                const cs = window.getComputedStyle(node)
                if ((cs.fontWeight === '700' || cs.fontWeight === 'bold') && node.textContent.trim()) hasBold = true
                if (cs.fontStyle === 'italic' && node.textContent.trim()) hasItalic = true
            }
            window._testHelper.assert('3.3c-bold-italic-bold-leaves-italic', !hasBold && hasItalic,
                'hasBold=' + hasBold + ' hasItalic=' + hasItalic + '. outerHTML=' + p.outerHTML)
        }, 100)
    }, 100)
}, 100)
```

Verify via `list_console_messages()` that all `3.3*` tests show PASS.

### Task 3.5: Content verification and undo/redo tests (Dimensions 6 + 7)

**File**: No source file — MCP `evaluate_script` tests only.

**Decision**: D-09

**Description**:
Verify DOM structure before/after style application and verify undo reverses then redo restores.

**Test 3.5-a: DOM gains bold markup after bold apply**:

```javascript
const p = window._testHelper.resetEditor('Content verification test')
window._testHelper.activateEditor()
const textNode = p.childNodes[0]

const childCountBefore = p.childNodes.length

window._testHelper.selectRange(textNode, 8, 20)
window._testHelper.clickBold()

setTimeout(() => {
    const childCountAfter = p.childNodes.length
    const hasBoldMark = p.querySelector('[style]') !== null || p.querySelector('strong') !== null || p.querySelector('b') !== null
    const changed = childCountAfter !== childCountBefore || hasBoldMark
    window._testHelper.assert('3.5a-dom-has-bold-markup', changed && hasBoldMark,
        'childsBefore=' + childCountBefore + ' childsAfter=' + childCountAfter + ' hasBoldMark=' + hasBoldMark)
}, 100)
```

**Test 3.5-b: Bold remove cleans up styled spans**:

```javascript
const { editorDiv } = window._testHelper.getEditor()
while (editorDiv.firstChild) editorDiv.removeChild(editorDiv.firstChild)
const p = document.createElement('p')
const span = document.createElement('span')
span.style.fontWeight = 'bold'
span.textContent = 'bold content here'
p.appendChild(span)
editorDiv.appendChild(p)

window._testHelper.clickBold()

setTimeout(() => {
    const styledCount = p.querySelectorAll('[style*="font-weight"]').length
    const strongCount = p.querySelectorAll('strong, b').length
    const cleaned = styledCount === 0 && strongCount === 0
    window._testHelper.assert('3.5b-dom-bold-removed', cleaned,
        'styledCount=' + styledCount + ' strongCount=' + strongCount + '. outerHTML=' + p.outerHTML)
}, 100)
```

**Test 3.5-c: Undo reverses bold apply**:

```javascript
const p = window._testHelper.resetEditor('Undo test text')
window._testHelper.activateEditor()
const textNode = p.childNodes[0]
const { sr } = window._testHelper.getEditor()
const editorDiv = sr.querySelector('[data-editor-root]')

// Apply bold
window._testHelper.selectRange(textNode, 0, 4)
window._testHelper.clickBold()

const outerHTMLAfterBold = p.outerHTML

// Wait for debounce to fire (300ms + margin = 400ms), then undo
setTimeout(() => {
    editorDiv.dispatchEvent(new KeyboardEvent('keydown', {
        key: 'z', ctrlKey: true, bubbles: true, composed: true, cancelable: true
    }))

    setTimeout(() => {
        // After undo, the bold span should be gone from the editor
        const boldSpans = editorDiv.querySelectorAll('[style*="font-weight"]')
        const strongEls = editorDiv.querySelectorAll('strong, b')
        const undone = boldSpans.length === 0 && strongEls.length === 0
        window._testHelper.assert('3.5c-undo-reverses-bold', undone,
            'boldSpans=' + boldSpans.length + ' strongEls=' + strongEls.length)
    }, 100)
}, 400)
```

**Test 3.5-d: Redo restores bold after undo**:

```javascript
const p = window._testHelper.resetEditor('Redo test text')
window._testHelper.activateEditor()
const textNode = p.childNodes[0]
const { sr } = window._testHelper.getEditor()
const editorDiv = sr.querySelector('[data-editor-root]')

// Apply bold

window._testHelper.clickBold()

// Undo after debounce
setTimeout(() => {
    editorDiv.dispatchEvent(new KeyboardEvent('keydown', {
        key: 'z', ctrlKey: true, bubbles: true, composed: true, cancelable: true
    }))

    // Redo
    setTimeout(() => {
        editorDiv.dispatchEvent(new KeyboardEvent('keydown', {
            key: 'y', ctrlKey: true, bubbles: true, composed: true, cancelable: true
        }))

        setTimeout(() => {
            const hasBold = window._testHelper.hasBoldInElement(editorDiv)
            window._testHelper.assert('3.5d-redo-restores-bold', hasBold, 'bold not restored after redo')
        }, 100)
    }, 100)
}, 400)
```

Verify via `list_console_messages()` that all `3.5*` tests show PASS.
