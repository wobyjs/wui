```typescript
  if (debounceTimer) clearTimeout(debounceTimer);
  debounceTimer = setTimeout(() => {
    const sel = getSelection();
    const selStart = sel ? $(sel).startOffset : 0;
    const selEnd = sel ? $(sel).endOffset : 0;
    const surface = document.querySelector('.editor-surface');
    if (!surface) return;

    // Trim forward history when saving new state
    if (currentIndex < historyStack.length - 1) {
      historyStack = historyStack.slice(0, currentIndex + 1);
    }

    historyStack.push({ html, selStart, selEnd, timestamp: Date.now() });
    if (historyStack.length > MAX_STACK) {
      historyStack.shift();
    } else {
      currentIndex++;
    }
  }, DEBOUNCE_MS);
}
```

```typescript
function restoreEditorState(state: HistoryState) {
  const surface = document.querySelector('.editor-surface');
  if (!surface || !state) return;
  surface.innerHTML = state.html;
  // Restore selection after DOM update
  requestAnimationFrame(() => {
    const sel = saveSelection();
    if (sel) {
      // Seek to approximate position using saved offsets
      const walker = document.createTreeWalker(surface, NodeFilter.SHOW_TEXT);
      let offset = 0;
      let targetNode: Text | null = null;
      let startOffset = state.selStart;
      let endOffset = state.selEnd;

      while (walker.nextNode()) {
        const node = walker.currentNode as Text;
        const len = node.textContent?.length || 0;
        if (offset + len >= startOffset && !targetNode) {
          targetNode = node;
          startOffset = startOffset - offset;
        }
        if (offset + len >= endOffset) {
          endOffset = endOffset - offset;
          break;
        }
        offset += len;
      }
      if (targetNode) {
        const range = document.createRange();
        try {
          range.setStart(targetNode, Math.min(startOffset, targetNode.textContent?.length || 0));
          range.collapse(true);
          const sel = window.getSelection();
          sel?.removeAllRanges();
          sel?.addRange(range);
        } catch {
          // Selection restore failed - cursor at end is acceptable
        }
      }
    }
  });
}
```

```typescript
    normalizeDOM = mod.normalizeDOM;
    mergeTextNodes = mod.mergeTextNodes;
    removeEmptySpans = mod.removeEmptySpans;

    container.contentEditable = 'true';
    document.body.appendChild(container);

    document.body.removeChild(container);
  });

  describe('mergeTextNodes', () => {
    it('merges adjacent text nodes in single paragraph', () => {
      container.innerHTML = '<p>ABC</p>';
      const p = container.querySelector('p')!;
      // Simulate adjacent text nodes by creating them
      while (p.firstChild) p.removeChild(p.firstChild);
      p.appendChild(document.createTextNode('AB'));
      p.appendChild(document.createTextNode('C'));

      mergeTextNodes(container);

      expect(p.childNodes.length).toBe(1);
      expect(p.textContent).toBe('ABC');

    it('merges text nodes separated by empty span', () => {
      container.innerHTML = '<p>AB<span></span>C</p>';
      mergeTextNodes(container);
      // Should merge text across empty span

      expect(textNodes).toBeLessThanOrEqual(3);

      container.innerHTML = '<p>AB</p><p>CD</p>';

      expect(ps[1].childNodes.length).toBe(1);
```
